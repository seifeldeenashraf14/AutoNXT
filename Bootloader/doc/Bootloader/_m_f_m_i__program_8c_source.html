48void MFMI_voidEraseSector (u8 Copy_u8SectorIndex)
49{
50 if ((MFMI_SECTOR5 < Copy_u8SectorIndex) || (MFMI_SECTOR0 == Copy_u8SectorIndex))
51 {
52 /* Do nothing */
53 }
54 else
55 {
56 /* Check if there is a Flash memory operation ongoing */
57 while (GET_BIT(FMI->SR, SR_BSY));
58
59 /* Check if the operations was locked */
60 if (GET_BIT(FMI->CR, CR_LOCK))
61 {
62 /* Unlock sequence */
63 FMI->KEYR = FMI_KEY1;
64 FMI->KEYR = FMI_KEY2;
65 }
66
67 /* Set the erase sector */
68 FMI->CR &= ~(MASK_4BIT << CR_SNB);
69 FMI->CR |= (Copy_u8SectorIndex << CR_SNB);
70 /* Enable sector erase */
71 SET_BIT(FMI->CR, CR_SER);
72 /* Enable start but */
73 SET_BIT(FMI->CR, CR_STRT);
74 /* Check if there is a Flash memory operation ongoing */
75 while (GET_BIT(FMI->SR, SR_BSY));
76 /* Clear end of operations */
77 SET_BIT(FMI->SR, SR_EOP);

101void MFMI_voidFlashWrite (u32 Copy_u32Address, u16 *Pointer_u16Arr, u8 Copy_u8DataLength)
102{
103 if ((0x0803FFFF < Copy_u32Address) || (0x08004000 > Copy_u32Address))
104 {
105 /* Do nothing */
106 }
107 else
108 {
109 u8 Local_u8Iterator;
110 /* Check if there is a Flash memory operation ongoing */
111 while (GET_BIT(FMI->SR, SR_BSY));
112 /* Check if the operations was locked */
113 if (GET_BIT(FMI->CR, CR_LOCK))
114 {
115 /* Unlock sequence */
116 FMI->KEYR = FMI_KEY1;
117 FMI->KEYR = FMI_KEY2;
118 }
119 /* Set programming size */
120 FMI->CR &= ~(MASK_2BIT << CR_PSIZE);
121 FMI->CR |= (MFMI_PSIZE << CR_PSIZE);
122
123 for (Local_u8Iterator = 0; Local_u8Iterator < Copy_u8DataLength; Local_u8Iterator++)
124 {
125 /* Enable programming */
126 SET_BIT(FMI->CR, CR_PG);
127 /* Program the address */
128 #if MFMI_PSIZE_BYTE == MFMI_PSIZE
129 *((volatile u8 *)Copy_u32Address) = Pointer_u16Arr[Local_u8Iterator];
130 Copy_u32Address += 1;
131 #elif MFMI_PSIZE_HALF_WORD == MFMI_PSIZE
132 *((volatile u16 *)Copy_u32Address) = Pointer_u16Arr[Local_u8Iterator];
133 Copy_u32Address += 2;
134 #elif MFMI_PSIZE_WORD == MFMI_PSIZE
135 *((volatile u32 *)Copy_u32Address) = Pointer_u16Arr[Local_u8Iterator];
136 Copy_u32Address += 4;
137 #elif MFMI_PSIZE_DOUBLE_WORD == MFMI_PSIZE
138 *((volatile u64 *)Copy_u32Address) = Pointer_u16Arr[Local_u8Iterator];
139 Copy_u32Address += 8;
140 #else
141 #error "Wrong PSIZE configuration, check include/MCAL/FMI/MFMI_config.h"
142 #endif
143 /* Check if there is a Flash memory operation ongoing */
144 while (GET_BIT(FMI->SR, SR_BSY));
145 /* Clear end of operations */
