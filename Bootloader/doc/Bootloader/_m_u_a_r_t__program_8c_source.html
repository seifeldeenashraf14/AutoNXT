24void MUART1_voidInit (void)
25{
26 /* Set Baud rate value */
27 UART1->BRR = MUART1_BRR_VALUE;
28 /* Set Oversampling value */
29 UART1->CR1.OVER8 = MUART1_OVERSAMPLING;
30 /* Set Data length */
31 UART1->CR1.M = MUART1_DATA_LENGTH;
32 /* Parity mode */
33 UART1->CR1.PCE = MUART1_PARITY_MODE;
34#if MUART_ENABLE == MUART1_PARITY_MODE
35 UART1->CR1.PS = MUART1_PARITY_SELECTION;
36#endif
37 /* Transmission complete interrupt */
38 UART1->CR1.TCIE = MUART1_TRANSMISSION_COMPELETE_INTERRUPT;
39 /* Read data register interrupt */
40 UART1->CR1.RXNEIE = MUART1_READ_DATA_REGISTER_EMPTY_INTERRUPT;
41 /* SEt stop bits */
42 UART1->CR2 &= ~(MASK_2BITS << CR2_STOP0);
43 UART1->CR2 |= (MUART1_STOP_BITS << CR2_STOP0);
44
45 UART1->SR = 0;
46}
