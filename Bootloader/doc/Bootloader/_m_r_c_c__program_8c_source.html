27void MRCC_voidInit (void)
28{
29 #if SYS_CLK == MRCC_HSI
30
31 /* Set HSITRIM (High speed internal clock trimming) default value */
32 RCC->CR &= ~(HSITRIM_MASK);
33 RCC->CR |= HSITRIM_MASK;
34
35 /* Enable HSI and Disable others */
36 CLR_BIT(RCC->CR, CR_HSE_ON);
37 CLR_BIT(RCC->CR, CR_PLL_ON);
38 SET_BIT(RCC->CR, CR_HSI_ON);
39
40 /* Polling on HSI ready flag */
41 while (!GET_BIT(RCC->CR, CR_HSI_RDY));
42
43 /* Switch to HSI */
44 RCC->CFGR &= ~(TWO_BITS_MASK << CFGR_SW0);
45 RCC->CFGR |= SW_HSI;
46
47 #elif SYS_CLK == MRCC_HSE
48 /* Check The clock source of HSE */
49 #if HSE_SOURCE == MRCC_HSE_CRYSTAL
50 /* Disable Bypass PIN */
51 CLR_BIT(RCC->CR, CR_HSE_BYP);
52 #elif HSE_SOURCE == MRCC_HSE_RC
53 /* Enable Bypass PIN */
54 SET_BIT(RCC->CR, CR_HSE_BYP);
55 #else
56 #error "Check HSE Clock source in MRCC_conf.h"
57 #endif
58 /* Enable HSE and Disable others */
59 CLR_BIT(RCC->CR, CR_PLL_ON);
60 CLR_BIT(RCC->CR, CR_HSI_ON);
61 SET_BIT(RCC->CR, CR_HSE_ON);
62
63 /* Polling on HSE ready flag */
64 while (!GET_BIT(RCC->CR, CR_HSE_RDY));
65
66 /* Switch to HSE */
67 RCC->CFGR &= ~(TWO_BITS_MASK << CFGR_SW0);
68 RCC->CFGR |= SW_HSE;
69
70 /* Clock Security System */
71 #if CLK_SECURITY == MRCC_CLK_SECURITY_ENABLE
72 SET_BIT(RCC->CR, CR_CSS_ON);
73 #elif CLK_SECURITY == MRCC_CLK_SECURITY_DISABLE
74 CLR_BIT(RCC->CR, CR_CSS_ON);
75 #else
76 #error "Check Clock security in MRCC_conf.h"
77 #endif
78
79 #elif SYS_CLK == MRCC_PLL
80
81 /* Set PLL Clock Source */
82 #if PLL_SOURCE == MRCC_PLL_HSI
83 /* Set PLL Source to HSI */
84 CLR_BIT(RCC->PLLCFGR, PLLCFGR_PLLCFGR);
85 #elif PLL_SOURCE == MRCC_PLL_HSE
86 /* Check The clock source of HSE */
87 #if HSE_SOURCE == MRCC_HSE_CRYSTAL
88 /* Disable Bypass PIN */
89 CLR_BIT(RCC->CR, CR_HSE_BYP);
90 #elif HSE_SOURCE == MRCC_HSE_RC
91 /* Enable Bypass PIN */
92 SET_BIT(RCC->CR, CR_HSE_BYP);
93 #else
94 #error "Check PLL HSE Clock source in MRCC_conf.h"
95 #endif
96 /* Set PLL Source to HSE */
97 SET_BIT(RCC->PLLCFGR, PLLCFGR_PLLCFGR);
98 #else
99 #error "Check PLL Clock Source in MRCC_conf.h"
100 #endif
101
102 /* Set PLL Prescaler */
103 RCC->PLLCFGR &= ~(TWO_BITS_MASK << PLLCFGR_PLLP0);
104 RCC->PLLCFGR |= (PLL_PRESCALER << PLLCFGR_PLLP0);
105
106 /* Set PLL Multiplication Factor */
107 RCC->PLLCFGR &= ~(NINE_BITS_MASK << PLLCFGR_PLLN);
108 RCC->PLLCFGR |= (PLL_MULTIPLICATION_FACTOR << PLLCFGR_PLLN);
109
110 /* Enable PLL and Disable others */
111 CLR_BIT(RCC->CR, CR_HSI_ON);
112 CLR_BIT(RCC->CR, CR_HSE_ON);
113 SET_BIT(RCC->CR, CR_PLL_ON);
114
115 /* Polling on PLL ready flag */
116 while (!GET_BIT(RCC->CR, CR_PLL_RDY));
117
118 /* Switch to PLL */
119 RCC->CFGR &= ~(TWO_BITS_MASK << CFGR_SW0);
120 RCC->CFGR |= SW_PLL;
121
122 #else
123 #error "Check System Clock source in MRCC_conf.h"
124 #endif
125}
133void MRCC_voidEnablePeripheral (u32 Copy_u32BusID, u32 Copy_u32PeripheralID)
134{
135 if (Copy_u32PeripheralID < 32)
136 {
137 switch (Copy_u32BusID)
138 {
139 /* Enable peripheral in APB1 bus */
140 case MRCC_APB1: SET_BIT(RCC->APB1ENR, Copy_u32PeripheralID); break;
141 /* Enable peripheral in APB2 bus */
142 case MRCC_APB2: SET_BIT(RCC->APB2ENR, Copy_u32PeripheralID); break;
143 /* Enable peripheral in AHB1 bus */
144 case MRCC_AHB1: SET_BIT(RCC->AHB1ENR, Copy_u32PeripheralID); break;
145 /* Enable peripheral in AHB2 bus */
146 case MRCC_AHB2: SET_BIT(RCC->AHB2ENR, Copy_u32PeripheralID); break;
147 default: break;
148 }
149 }
150 else
151 {
152 /* Do nothing */
