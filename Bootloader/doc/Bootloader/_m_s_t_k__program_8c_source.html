244{
245 u8 Local_u8Temp;
246 /* Check if the Notification has valid address */
247 if (Global_pfNotification != NULL)
248 {
249 /* Execute the function */
250 Global_pfNotification();
251 /* Clear interrupt flag */
252 Local_u8Temp = GET_BIT(CTRL, CTRL_COUNTFLAG);
253 /* Check if it was single interval or periodic */
254 if (SINGLE_INTERVAL == Global_u8IntervalMode)
